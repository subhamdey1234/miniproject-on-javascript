Execute factorials of Number
     

    function factorial(n) {
        let f=1;
        for(let i=1;i<=n;i++){
        f=f*i;
    }
    return f;
    }

  function  printfact() {
  let n=Number(prompt("Enter the Number=>"));
  alert("The Factorial of "+n+" is=>"+factorial(n));
  }      
     



Execute Recursive Factorial
     

    function recfactorial(n)
    {
        if (n == 0)
            return 1;
        return n * factorial(n - 1);
    }

Execute Ranged Factorial
     

    function rangedfactorial() {
        let ll=Number(prompt("Enter the Lower Limit=>"));
        let ul=Number(prompt("Enter the Upper Limit=>"));
        let prod=1;
        for(let i=ll;i<=ul;i++)
    {
        prod=prod*i;
    }
    alert("ranged factorial=>"+prod);
    }
