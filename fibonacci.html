Execute Range Of Fibonacci Series



    function fibo() {
        let a = 0;
        let b = 1;
        let c = 0;

        let ul = Number(prompt("Enter the upper limit"));
        let ll = Number(prompt("Enter the lower limit"));
        if (ll == 0) {
            alert("Number => " + a);
            console.log(a);
        }

        for (;;) {
            c = a + b;
            if (c > ul) {
                break;
            }
            if (c >= ll) {
                alert("Number => " + c);
                console.log(c);
            }
            a = b;
            b = c;
        }
    }

    function nextfibo() {
        let a = 0;
        let b = 1;
        let c = 0;

        let ul = Number(prompt("Enter the Lucky Number"));

        if (ul === 0) {
            alert("Next Fibonacci number => " + a);
            console.log(a);
            return;
        }

        while (c < ul) {
            c = a + b;
            a = b;
            b = c;
        }
