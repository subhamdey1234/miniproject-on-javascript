Perfect Numbers Programs

       
Execute Perfect Number 1
     

    function perfect(n) {
        let sum=0;
        for(let i=1;i<=n/2;i++)
    {
    if (n%i===0) {
        sum+=i;
    }
    
    }
    if (sum==n && sum!=0) {
        console.log("The Number is a Perfect Number");
        return true;
        
    }
    else{
        console.log("The number is not a Perfect Number");
        return false;
    }
    }   
     



Execute alternate Perfect Number
     

    function alternateperfect() {
        let n=Number(prompt("Enter The Ending Number=>"));
        let s=Number(prompt("Enter The  Starting Number=>"));
        let c=0;
        for(let i=s;i<=n;i++)
    {
    if (perfect(i)) {
        c++;
        if (c%2!=0) {
            alert("The Alternate Perfect Numbers=>"+i);
        }
    }
    }
    }
     




Execute First Three Perfect Number
     

    function firstthreeperfect() {
        let n=Number(prompt("Enter The Ending Number=>"));
        let s=Number(prompt("Enter The  Starting Number=>"));
        let c=0;
        for(let i=s;i<=n;i++)
    {
    if (perfect(i)) {
        c++;
        alert("The Alternate Perfect Numbers=>"+i);
        if (c==3) {
            break;
        }
    }
    }
    }





Execute Last Three Perfect Number
     

    function lastthreeperfect() {
        let n=Number(prompt("Enter The Ending Number=>"));
        let s=Number(prompt("Enter The  Starting Number=>"));
        let c=0;
        for(let i=n;i>=s;i--)
    {
    if (perfect(i)) {
        c++;
        alert("The Last Three Perfect Numbers=>"+i);
        if (c==3) {
            break;
        }
    }
    }
