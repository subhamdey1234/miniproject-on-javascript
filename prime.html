Prime Number 1
     

    function prime(n) {
        if (n<2) {
          return false;
        }  
        for(let i=2;i<=Math.trunc(n/2);i++)
        {
          if (n%i==0) {
              return false;
          }
          
        }
        return true;
      }
   console.log(prime(Number(prompt("Enter the number==>")))
